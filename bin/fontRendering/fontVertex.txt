#version 330

in vec2 position;
in vec2 textureCoords;

out vec2 pass_textureCoords;

uniform vec2 translation;

void main(void){
	float offset = 1.0;
	gl_Position = vec4(position.x + translation.x * offset, position.y + ((translation.y+1.0) * - offset), 0.0, 1.0);
	pass_textureCoords = textureCoords;
}
